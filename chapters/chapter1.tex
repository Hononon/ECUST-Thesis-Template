\section{算法示例}
\begin{algorithm}[H]
  \caption{示例算法}
  \label{alg:template}
  \begin{algorithmic}[1]
    \Require 输入向量 $\mathbf{x}$
    \Ensure 输出结果 $y$
    \State $y \gets 0$
    \For{each $x_i$ in $\mathbf{x}$}
    \State $y \gets y + x_i$
    \EndFor
    \State \Return $y$
  \end{algorithmic}
\end{algorithm}
